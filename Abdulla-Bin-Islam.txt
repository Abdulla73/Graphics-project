bool isNight = false;
///display-functions
void display()
{
    glClear(GL_COLOR_BUFFER_BIT);


   if (isNight)
    {
        Night_Sky();
        Night_Background();
        Night_Hill();
        D_Building();
        Annex7();
        Annex6();
        Night_Tree1();
        Night_Tree2();
        Moon();
        Night_Field();

    }
   else
    {
        Sky();
        Sun();
        Background();
        Hill();
        Cloud1();
        Cloud2();
        Cloud3();
        D_Building();
        Annex7();
        Annex6();
        Tree1();
        Tree2();
        Field();

    }
   Road();
    glFlush();
    glutSwapBuffers();
}

void handleButton(int button, int state, int x, int y) {
    if (button == GLUT_LEFT_BUTTON && state == GLUT_DOWN)
    {

        isNight = !isNight;
        glutPostRedisplay();
    }
}
